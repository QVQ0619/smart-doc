# 立项审查AI辅助系统 — 前端骨架 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 搭建立项审查AI辅助系统的三栏前端骨架（左菜单 / 中占位工作区 / 右真实 Blade 对话），工程化、可维护、可逐页扩展。

**Architecture:** React 19 + Vite SPA。zustand 管路由，按 route key 切换中间占位页；右栏用 `@blade-hq/agent-kit` 的 `ChatView` 直连 Blade Agent（`http://115.190.152.1:8020`），token 走 `.env`。无后端、无 react-router。

**Tech Stack:** React 19, Vite 5, TypeScript, Ant Design 5, zustand 5, @tanstack/react-query 5, @blade-hq/agent-kit 0.5.11, sonner；测试用 Vitest + @testing-library/react + jsdom。

## Global Constraints

- React 必须是 19，不支持 18。
- `@blade-hq/agent-kit` 固定 `0.5.11`，**按子入口导入**（`/client` `/react` `/chat`），不从包根 `@blade-hq/agent-kit` 导入，不存在的方法/导出不要用。
- 安装依赖不得用 `--force` / `--legacy-peer-deps`。
- Blade `baseUrl` 必须是 origin，**不带 pathname**（`http://115.190.152.1:8020`）。
- `VITE_BLADE_TOKEN` 默认留空，由使用者填 `sk-blade-v2-...`；token 用 getter 形式注入。
- ChatView 外层容器必须满足：`height` + `min-height:0` + `display:flex` + `flex-direction:column` + `overflow:hidden`。
- zustand selector 不得每次返回新数组/对象（React 19 下会无限渲染），用稳定常量。
- 视觉浅色专业风：白/浅灰底 + 蓝主色（`#1677ff`），卡片化，对齐 demo。
- 包管理器统一用 `npm`。所有命令在项目根 `E:\shixi\smart-doc` 执行。

- [ ] **Step 3: 写实现 src/blade/config.ts**

```ts
const DEFAULT_BASE_URL = "http://115.190.152.1:8020";

export function getBaseUrl(): string {
  const raw = import.meta.env.VITE_BLADE_API_BASE?.trim();
  const base = raw && raw.length > 0 ? raw : DEFAULT_BASE_URL;
  return base.replace(/\/+$/, "");
}

export function getToken(): string {
  return import.meta.env.VITE_BLADE_TOKEN?.trim() ?? "";
}

export function hasToken(): boolean {
  return getToken().length > 0;
}

export function getSolutionId(): string | undefined {
  const v = import.meta.env.VITE_BLADE_SOLUTION_ID?.trim();
  return v && v.length > 0 ? v : undefined;
}

export function getBizRoleId(): string | undefined {
  const v = import.meta.env.VITE_BLADE_BIZ_ROLE_ID?.trim();
  return v && v.length > 0 ? v : undefined;
}
```

import { useCallback, useEffect, useState } from "react";
import { Button, Tooltip } from "antd";
import { PlusOutlined } from "@ant-design/icons";
import { ChatView } from "@blade-hq/agent-kit/chat";
import { useSessionStore } from "@blade-hq/agent-kit/react";
import { bladeClient } from "../blade/client";
import { hasToken, getSolutionId, getBizRoleId } from "../blade/config";

export default function ChatPanel() {
  const [sessionId, setSessionId] = useState<string | null>(null);
  const [creating, setCreating] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const tokenReady = hasToken();

  const createSession = useCallback(async () => {
    setCreating(true);
    setError(null);
    try {
      const { session_id } = await bladeClient.sessions.createSession(
        "立项审查对话",
        { solution_id: getSolutionId(), biz_role_id: getBizRoleId() },
      );
      setSessionId(session_id);
      useSessionStore.getState().setActiveSession(session_id);
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    } finally {
      setCreating(false);
    }
  }, []);

  useEffect(() => {
    if (tokenReady && !sessionId && !creating) {
      void createSession();
    }
    // 仅在 token 就绪状态变化时触发一次自动建会话
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [tokenReady]);

  return (
    <>
      <div
        style={{
          display: "flex",
          alignItems: "center",
          justifyContent: "space-between",
          padding: "12px 16px",
          borderBottom: "1px solid #e5e6eb",
          flex: "0 0 auto",
        }}
      >
        <strong>对话助手</strong>
        <Tooltip title="新建会话">
          <Button
            size="small"
            icon={<PlusOutlined />}
            disabled={!tokenReady || creating}
            onClick={() => void createSession()}
          >
            新建会话
          </Button>
        </Tooltip>
      </div>

      <div style={{ flex: "1 1 auto", minHeight: 0, display: "flex", flexDirection: "column", overflow: "hidden" }}>
        {!tokenReady ? (
          <div style={{ padding: 24, color: "#86909c", fontSize: 13 }}>
            未配置 Blade API Key。请在项目根的 <code>.env</code> 中填写{" "}
            <code>VITE_BLADE_TOKEN=sk-blade-v2-...</code>，保存后重启 <code>npm run dev</code>。
          </div>
        ) : error ? (
          <div style={{ padding: 24, color: "#f53f3f", fontSize: 13 }}>
            建会话失败：{error}
            <div style={{ marginTop: 12 }}>
              <Button size="small" onClick={() => void createSession()}>
                重试
              </Button>
            </div>
          </div>
        ) : sessionId ? (
          <ChatView sessionId={sessionId} />
        ) : (
          <div style={{ padding: 24, color: "#86909c", fontSize: 13 }}>
            正在创建会话…
          </div>
        )}
      </div>
    </>
  );
}

```ts
import { create } from "zustand";

const KEY = "chat-panel-collapsed";

function readInitial(): boolean {
  try {
    return localStorage.getItem(KEY) === "1";
  } catch {
    return false;
  }
}

interface ChatCollapseState {
  collapsed: boolean;
  toggle: () => void;
}

export const useChatCollapseStore = create<ChatCollapseState>((set, get) => ({
  collapsed: readInitial(),
  toggle: () => {
    const next = !get().collapsed;
    try {
      localStorage.setItem(KEY, next ? "1" : "0");
    } catch {
      /* localStorage 不可用时仅在内存切换 */
    }
    set({ collapsed: next });
  },
}));
```
