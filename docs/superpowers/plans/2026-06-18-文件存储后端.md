# 文件存储后端（MySQL）Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为立项审查前端新增 FastAPI 后端，接入本地 MySQL `smart`，把上传文件落盘 + 元数据入库，规则文件与申请材料分类隔离，并在「规则库」「审查文档库」两页提供上传/列表/下载/删除。

**Architecture:** FastAPI(:8000) 作中间层；前端(:3200)经 Vite dev proxy 调 `/api`。文件存磁盘 `storage/uploads/<category>/`，元数据存 MySQL `smart.documents`。后端用依赖注入（DB session、存储服务、上传上限）以便测试用 SQLite 内存库 + 临时目录隔离。

**Tech Stack:** Python 3.13、FastAPI、SQLAlchemy 2.0、PyMySQL、pydantic-settings、python-multipart、uvicorn、pytest+httpx；前端 React 19 + Ant Design + @tanstack/react-query + sonner（已装）。

## Global Constraints

- Python 3.13；后端用 venv：`backend/.venv`，所有后端命令用 `.venv/Scripts/python`（Windows）。
- MySQL 连接：`mysql+pymysql://root:root@localhost:3306/smart`；`smart` 库须预先存在（`CREATE DATABASE smart CHARACTER SET utf8mb4;`），表由应用启动 `Base.metadata.create_all` 自动建。
- 类别合法值恒为 `("rule", "application")`，非法值接口返回 400。
- 文件存盘路径 `storage/uploads/<category>/<uuid4hex>.<ext>`；元数据出参 `DocumentOut` 不暴露 `stored_name`/`rel_path`。
- 单文件上限默认 50MB（`SMART_MAX_UPLOAD_MB`）；超限文件记入 `failed`，不落盘不入库，不影响同批其它文件；上传整体返回 HTTP 200 + `{uploaded, failed}`。
- 前端：Vite proxy `/api` → `http://localhost:8000`；复用已装的 react-query；通知用 sonner `toast`（不用 antd 静态 message，避免 console 警告）。
- 测试输出必须 pristine（无警告）；前端在异步事件后用 `findBy*`/`waitFor` 断言。现有 16 个前端测试不回归。
- 后端命令 CWD = `E:\shixi\smart-doc\backend`；前端命令 CWD = `E:\shixi\smart-doc`。

- [ ] **Step 5: 写 storage.py**

`backend/app/storage.py`:
```python
import hashlib
import uuid
from dataclasses import dataclass
from pathlib import Path


@dataclass
class StoredFile:
    stored_name: str
    rel_path: str
    size_bytes: int
    sha256: str


class FileTooLargeError(Exception):
    def __init__(self, limit_bytes: int):
        self.limit_bytes = limit_bytes
        super().__init__(f"file exceeds {limit_bytes} bytes")


class Storage:
    CHUNK = 1024 * 1024

    def __init__(self, base_dir):
        self.base_dir = Path(base_dir)

    def save(self, category: str, original_name: str, stream, max_bytes: int) -> StoredFile:
        ext = Path(original_name).suffix
        stored_name = f"{uuid.uuid4().hex}{ext}"
        rel_path = f"{category}/{stored_name}"
        dest = self.base_dir / category / stored_name
        dest.parent.mkdir(parents=True, exist_ok=True)

        hasher = hashlib.sha256()
        size = 0
        try:
            with dest.open("wb") as f:
                while True:
                    chunk = stream.read(self.CHUNK)
                    if not chunk:
                        break
                    size += len(chunk)
                    if size > max_bytes:
                        raise FileTooLargeError(max_bytes)
                    hasher.update(chunk)
                    f.write(chunk)
        except Exception:
            dest.unlink(missing_ok=True)
            raise

        return StoredFile(stored_name, rel_path, size, hasher.hexdigest())

    def delete(self, rel_path: str) -> None:
        (self.base_dir / rel_path).unlink(missing_ok=True)


def get_storage() -> Storage:
    from .config import settings

    return Storage(settings.storage_dir)
```

```python
from fastapi import APIRouter, Depends, File, Form, HTTPException, UploadFile
from sqlalchemy.orm import Session

from ..config import get_max_upload_bytes
from ..db import get_db
from ..models import Document, VALID_CATEGORIES
from ..schemas import DocumentOut, FailedItem, UploadResult
from ..storage import FileTooLargeError, Storage, get_storage

router = APIRouter(tags=["documents"])


def _check_category(category: str) -> None:
    if category not in VALID_CATEGORIES:
        raise HTTPException(status_code=400, detail=f"invalid category: {category}")


@router.post("/documents", response_model=UploadResult)
def upload_documents(
    category: str = Form(...),
    files: list[UploadFile] = File(...),
    db: Session = Depends(get_db),
    storage: Storage = Depends(get_storage),
    max_bytes: int = Depends(get_max_upload_bytes),
) -> UploadResult:
    _check_category(category)
    uploaded: list[DocumentOut] = []
    failed: list[FailedItem] = []

    for up in files:
        name = up.filename or "unnamed"
        try:
            stored = storage.save(category, name, up.file, max_bytes)
        except FileTooLargeError as e:
            failed.append(FailedItem(name=name, reason=f"超过 {e.limit_bytes} 字节上限"))
            continue
        except Exception as e:  # noqa: BLE001
            failed.append(FailedItem(name=name, reason=f"落盘失败: {e}"))
            continue

        doc = Document(
            category=category,
            original_name=name,
            stored_name=stored.stored_name,
            rel_path=stored.rel_path,
            size_bytes=stored.size_bytes,
            mime_type=up.content_type or "application/octet-stream",
            sha256=stored.sha256,
        )
        try:
            db.add(doc)
            db.commit()
            db.refresh(doc)
        except Exception as e:  # noqa: BLE001
            db.rollback()
            storage.delete(stored.rel_path)
            failed.append(FailedItem(name=name, reason=f"入库失败: {e}"))
            continue

        uploaded.append(DocumentOut.model_validate(doc))

    return UploadResult(uploaded=uploaded, failed=failed)
```

def list_documents(category: str, db: Session = Depends(get_db)) -> list[DocumentOut]:
    _check_category(category)
    rows = (
        db.execute(
            select(Document)
            .where(Document.category == category)
            .order_by(Document.created_at.desc(), Document.id.desc())
        )
        .scalars()
        .all()
    )
    return [DocumentOut.model_validate(r) for r in rows]


@router.get("/documents/{doc_id}/download")
def download_document(
    doc_id: int,
    db: Session = Depends(get_db),
    storage: Storage = Depends(get_storage),
):
    doc = db.get(Document, doc_id)
    if doc is None:
        raise HTTPException(status_code=404, detail="document not found")
    path = storage.base_dir / doc.rel_path
    if not path.exists():
        raise HTTPException(status_code=404, detail="file missing on disk")
    return FileResponse(path, filename=doc.original_name, media_type=doc.mime_type)


@router.delete("/documents/{doc_id}", status_code=204)
def delete_document(
    doc_id: int,
    db: Session = Depends(get_db),
    storage: Storage = Depends(get_storage),
):
    doc = db.get(Document, doc_id)
    if doc is None:
        raise HTTPException(status_code=404, detail="document not found")
    rel = doc.rel_path
    db.delete(doc)
    db.commit()
    storage.delete(rel)
```

export type DocumentCategory = "rule" | "application";

export interface DocumentOut {
  id: number;
  category: DocumentCategory;
  original_name: string;
  size_bytes: number;
  mime_type: string;
  sha256: string;
  created_at: string;
}

export interface UploadFailed {
  name: string;
  reason: string;
}

export interface UploadResult {
  uploaded: DocumentOut[];
  failed: UploadFailed[];
}

async function handle<T>(res: Response): Promise<T> {
  if (!res.ok) {
    const text = await res.text().catch(() => "");
    throw new Error(`HTTP ${res.status}: ${text || res.statusText}`);
  }
  if (res.status === 204) return undefined as T;
  return (await res.json()) as T;
}

export function listDocuments(category: DocumentCategory): Promise<DocumentOut[]> {
  return fetch(`/api/documents?category=${encodeURIComponent(category)}`).then((r) =>
    handle<DocumentOut[]>(r),
  );
}

export function uploadDocuments(
  category: DocumentCategory,
  files: File[],
): Promise<UploadResult> {
  const form = new FormData();
  form.append("category", category);
  for (const f of files) form.append("files", f);
  return fetch("/api/documents", { method: "POST", body: form }).then((r) =>
    handle<UploadResult>(r),
  );
}

export function deleteDocument(id: number): Promise<void> {
  return fetch(`/api/documents/${id}`, { method: "DELETE" }).then((r) => handle<void>(r));
}

export function downloadDocumentUrl(id: number): string {
  return `/api/documents/${id}/download`;
}
```

`src/components/DocumentManager.tsx`:
```tsx
import { useRef } from "react";
import { Button, Popconfirm, Space, Table } from "antd";
import { UploadOutlined } from "@ant-design/icons";
import type { ColumnsType } from "antd/es/table";
import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { toast } from "sonner";
import {
  listDocuments,
  uploadDocuments,
  deleteDocument,
  downloadDocumentUrl,
  type DocumentCategory,
  type DocumentOut,
} from "../api/documents";

function humanSize(bytes: number): string {
  if (bytes < 1024) return `${bytes} B`;
  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
  return `${(bytes / 1024 / 1024).toFixed(1)} MB`;
}

interface Props {
  category: DocumentCategory;
  title: string;
}

export default function DocumentManager({ category, title }: Props) {
  const qc = useQueryClient();
  const inputRef = useRef<HTMLInputElement>(null);
  const key = ["documents", category];

  const listQuery = useQuery({ queryKey: key, queryFn: () => listDocuments(category) });

  const uploadMut = useMutation({
    mutationFn: (files: File[]) => uploadDocuments(category, files),
    onSuccess: (res) => {
      if (res.uploaded.length) toast.success(`成功上传 ${res.uploaded.length} 个文件`);
      if (res.failed.length) {
        toast.warning(
          `${res.failed.length} 个失败：` +
            res.failed.map((f) => `${f.name}(${f.reason})`).join("，"),
        );
      }
      qc.invalidateQueries({ queryKey: key });
    },
    onError: (e) => toast.error(e instanceof Error ? e.message : String(e)),
  });

  const deleteMut = useMutation({
    mutationFn: (id: number) => deleteDocument(id),
    onSuccess: () => {
      toast.success("已删除");
      qc.invalidateQueries({ queryKey: key });
    },
    onError: (e) => toast.error(e instanceof Error ? e.message : String(e)),
  });

  function onPick(e: React.ChangeEvent<HTMLInputElement>) {
    const files = Array.from(e.target.files ?? []);
    if (files.length) uploadMut.mutate(files);
    e.target.value = "";
  }

  const columns: ColumnsType<DocumentOut> = [
    { title: "文件名", dataIndex: "original_name", key: "original_name" },
    { title: "大小", dataIndex: "size_bytes", key: "size_bytes", render: (b: number) => humanSize(b) },
    { title: "类型", dataIndex: "mime_type", key: "mime_type" },
    {
      title: "上传时间",
      dataIndex: "created_at",
      key: "created_at",
      render: (t: string) => new Date(t).toLocaleString(),
    },
    {
      title: "操作",
      key: "actions",
      render: (_: unknown, row: DocumentOut) => (
        <Space>
          <a href={downloadDocumentUrl(row.id)} target="_blank" rel="noreferrer">
            下载
          </a>
          <Popconfirm
            title="确认删除该文件？"
            okText="确定"
            cancelText="取消"
            onConfirm={() => deleteMut.mutate(row.id)}
          >
            <Button type="link" danger>
              删除
            </Button>
          </Popconfirm>
        </Space>
      ),
    },
  ];

  return (
    <div>
      <div style={{ marginBottom: 16 }}>
        <input ref={inputRef} type="file" multiple hidden onChange={onPick} />
        <Button
          type="primary"
          icon={<UploadOutlined />}
          loading={uploadMut.isPending}
          onClick={() => inputRef.current?.click()}
        >
          上传{title}
        </Button>
      </div>
      <Table
        rowKey="id"
        size="middle"
        loading={listQuery.isLoading}
        dataSource={listQuery.data ?? []}
        columns={columns}
        pagination={false}
      />
    </div>
  );
}
```
